# -------------------------

# -------------------------
# Ready (maybe public in your router; if protected, use header)
# -------------------------
Write-Host "`n--- GET /readyz ---"
try {
  Invoke-RestMethod "$BASE/readyz" | ConvertTo-Json -Depth 20
} catch {
  Write-Host "readyz failed without key. Trying with X-API-Key..."
  Invoke-RestMethod "$BASE/readyz" -Headers $H | ConvertTo-Json -Depth 20
